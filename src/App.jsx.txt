const fetchGeminiWithRetry = async (prompt, retries = 5) => {
  const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;
  const options = { 
    method: 'POST', 
    headers: { 'Content-Type': 'application/json' }, 
    body: JSON.stringify({ contents: [{ parts: [{ text: prompt }] }] }) 
  };
  const delays = [1000, 2000, 4000, 8000, 16000];
  
  for (let i = 0; i < retries; i++) {
    try {
      const response = await fetch(url, options);
      if (!response.ok) throw new Error(`HTTP error! status: ${response.status}`);
      const data = await response.json();
      return data.candidates?.[0]?.content?.parts?.[0]?.text || '';
    } catch (error) { 
      if (i === retries - 1) throw error; 
      await new Promise(res => setTimeout(res, delays[i])); 
    }
  }
  return "";
};

const openWhatsApp = (phone, message = "") => {
  if (!phone) return;
  const cleanPhone = String(phone).replace(/\D/g, '');
  window.open(`https://wa.me/${cleanPhone}${message ? `?text=${encodeURIComponent(message)}` : ''}`, '_blank');
};

const normalizeText = (text) => {
  if (!text) return "";
  return String(text).toLowerCase().normalize("NFD").replace(/[\u0300-\u036f]/g, "");
};

const safeFormatDate = (dateStr) => {
  try {
    const d = new Date(dateStr);
    if (isNaN(d.getTime())) return String(dateStr);
    return d.toLocaleDateString('es-ES', { weekday: 'short', day: 'numeric', month: 'short' });
  } catch (e) { return String(dateStr); }
};

const safeGetDay = (dateStr) => {
  try {
    const d = new Date(dateStr);
    if (isNaN(d.getTime())) return '--';
    return String(d.getDate());
  } catch (e) { return '--'; }
};

const PatientProfile = ({ patient, onBack, onAddHistory, onDelete, onScheduleAppointment }) => {
  const [sumLoading, setSumLoading] = useState(false);
  const [sum, setSum] = useState('');

  const generateAI = async () => {
    if (!patient?.histories?.length) return;
    setSumLoading(true);
    const text = patient.histories.map(h => `${h.date}: Dolor ${h.painLevel}. Notas: ${h.notes}`).join(' | ');
    const res = await fetchGeminiWithRetry(`Genera un resumen analítico y profesional muy breve (máx 3 líneas) de la evolución quiropráctica del paciente en base a estas notas clínicas: ${text}`);
    setSum(String(res)); 
    setSumLoading(false);
  };

  if (!patient) return null;

  let bmiValue = '--';
  if (patient.weight && patient.height) {
    const w = parseFloat(String(patient.weight).replace(',', '.'));
    const h = parseFloat(String(patient.height).replace(',', '.')) / 100;
    if (w > 0 && h > 0) {
      bmiValue = String((w / (h * h)).toFixed(1));
    }
  }

  return (
    <div className="bg-[#020617] min-h-full pb-20 animate-fade-in relative text-white">
      <div className="bg-gradient-to-b from-indigo-800 to-slate-950 text-white p-8 sm:p-12 pt-16 sm:pt-20 pb-20 rounded-b-[60px] sm:rounded-b-[80px] relative shadow-2xl border-b-[10px] border-indigo-900 text-center overflow-hidden">
        <button onClick={onBack} className="absolute top-8 left-6 sm:left-10 p-4 sm:p-5 bg-white/10 rounded-full border border-white/20 active:scale-90 transition"><ChevronRight className="w-6 h-6 sm:w-8 sm:h-8 rotate-180 text-cyan-400" /></button>
        <button onClick={onDelete} className="absolute top-8 right-6 sm:right-10 p-4 sm:p-5 bg-rose-500/20 rounded-full border border-rose-500/40 text-rose-500 active:scale-90 transition"><Trash2 className="w-6 h-6 sm:w-8 sm:h-8" /></button>
        
        <div className="flex flex-col items-center z-10 relative mt-6">
          <div className="w-32 h-32 sm:w-40 sm:h-40 rounded-[50px] bg-white p-2 flex items-center justify-center shadow-[0_0_80px_rgba(6,182,212,0.4)] mb-6 border-4 border-cyan-400">

const MarketingTab = ({ doctorInfo }) => {
  const [topic, setTopic] = useState('');
  const [loading, setLoading] = useState(false);
  const [promos, setPromos] = useState([]);

  const generatePromos = async () => {
    if (!topic) return;
    setLoading(true);
    const prompt = `Eres experto en copywriting quiropráctico. Escribe 3 mensajes persuasivos con emojis para vender en redes o WhatsApp. Motivo: "${topic}". Clínica: "${doctorInfo.clinic}". Dr: "${doctorInfo.name}". Devuelve ÚNICAMENTE un JSON con un array llamado "promociones" de 3 strings simples (solo texto, no objetos).`;
    try {
      const response = await fetchGeminiWithRetry(prompt);
      const jsonStr = String(response).replace(/```json|```/g, '').trim();
      const data = JSON.parse(jsonStr);
      if (data && Array.isArray(data.promociones)) {
        setPromos(data.promociones.map(p => typeof p === 'string' ? p : JSON.stringify(p)));
      } else {
        setPromos(["Error de formato IA."]);
      }
    } catch (error) { setPromos(["Error de conexión con la IA de Google."]); }
    finally { setLoading(false); }
  };

  return (
    <div className="p-4 sm:p-6 space-y-4 sm:space-y-6 animate-fade-in pb-10 text-white">
      <div className="bg-gradient-to-br from-violet-600 via-indigo-600 to-blue-700 rounded-[30px] sm:rounded-[40px] p-6 sm:p-10 text-white shadow-2xl border border-white/20 text-center sm:text-left">
        <h2 className="text-xl sm:text-3xl font-black mb-2 tracking-tighter flex items-center justify-center sm:justify-start gap-2 sm:gap-3"><Sparkles className="w-6 h-6 sm:w-10 sm:h-10 text-cyan-300 animate-pulse" /> MARKETING IA</h2>
        <p className="text-indigo-100 text-[8px] sm:text-[10px] font-black uppercase tracking-[0.3em] sm:tracking-[0.4em]">Creación Automática de Anuncios</p>
      </div>

      <div className="bg-indigo-900/40 backdrop-blur-md p-5 sm:p-8 rounded-[30px] sm:rounded-[40px] shadow-2xl border border-indigo-400/20 text-left">

        <textarea placeholder="Ej: Mes del adulto mayor, Chequeo postural deportivo..." className="w-full p-4 sm:p-6 bg-indigo-950 border-2 border-indigo-800 rounded-[20px] sm:rounded-[30px] outline-none font-bold text-white shadow-inner h-24 sm:h-32 focus:border-cyan-400 transition-all uppercase text-[10px] sm:text-sm" value={topic} onChange={(e) => setTopic(e.target.value)} />
        <button onClick={generatePromos} disabled={loading || !topic} className="w-full bg-gradient-to-r from-cyan-500 to-blue-600 text-black font-black py-4 sm:py-6 rounded-[20px] sm:rounded-[30px] shadow-[0_15px_30px_rgba(6,182,212,0.3)] mt-4 sm:mt-5 disabled:opacity-40 flex items-center justify-center gap-2 sm:gap-3 active:scale-95 transition-transform uppercase italic text-xs sm:text-base">
          {loading ? <Loader2 className="w-4 h-4 sm:w-6 sm:h-6 animate-spin" /> : <Megaphone className="w-4 h-4 sm:w-6 sm:h-6" />} Generar Copywriting
        </button>
      </div>

      <div className="space-y-4">
        {promos.map((promo, i) => {
          const textToRender = typeof promo === 'string' ? promo : JSON.stringify(promo);
          return (
            <div key={i} className="bg-white/5 backdrop-blur-md p-5 sm:p-8 rounded-[25px] sm:rounded-[35px] border border-white/10 relative text-left shadow-lg">
              <p className="text-[11px] sm:text-sm text-indigo-50 font-medium leading-relaxed mb-4 sm:mb-6 italic">"{textToRender}"</p>
              <button onClick={() => { navigator.clipboard.writeText(textToRender); }} className="px-4 sm:px-5 py-2 sm:py-2.5 bg-cyan-500/10 text-cyan-400 rounded-full text-[8px] sm:text-[9px] font-black uppercase border border-cyan-400/20 active:bg-cyan-500 active:text-black transition-all shadow-sm">Copiar Mensaje</button>

// --- COMPONENTE PRINCIPAL ---
export default function App() {
  const [showIntro, setShowIntro] = useState(true);
  const [user, setUser] = useState(null);
  const [loading, setLoading] = useState(true);
  const [authInProcess, setAuthInProcess] = useState(false);
  const [isBiometricNeeded, setIsBiometricNeeded] = useState(false);
  const [doctorInfo, setDoctorInfo] = useState({ name: 'Especialista', clinic: 'QuiroClínica Pro', modPassword: '' });
  const [activeTab, setActiveTab] = useState('home');
  const [selectedPatientId, setSelectedPatientId] = useState(null);
  const [isOnline, setIsOnline] = useState(true);
  
  const [patients, setPatients] = useState([]);
  const [appointments, setAppointments] = useState([]);

  const [modals, setModals] = useState({ appointment: false, patient: false, history: false, settings: false, password: false });
  const [pendingAction, setPendingAction] = useState(null);
  const [prefilledPatientId, setPrefilledPatientId] = useState("");

  const handleLogout = async () => {
    try {
      await signOut(auth);
      localStorage.removeItem('quiro_registered');
      setIsBiometricNeeded(false);
      setModals({ appointment: false, patient: false, history: false, settings: false, password: false });
    } catch (err) { console.error("Error Logout:", err); }
  };

  useEffect(() => {
    setIsOnline(navigator.onLine);
    const handleStatus = () => setIsOnline(navigator.onLine);
    window.addEventListener('online', handleStatus);
    window.addEventListener('offline', handleStatus);
    return () => {
      window.removeEventListener('online', handleStatus);
      window.removeEventListener('offline', handleStatus);
    };
  }, []);

  useEffect(() => {
    const initAuth = async () => {
      try {
        if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
          try {
            await signInWithCustomToken(auth, __initial_auth_token);
          } catch (e) {
            await signInAnonymously(auth);
          }
        }
        else {
          await signInAnonymously(auth);
        }
      } catch (err) { console.error("Init Auth:", err); }
    };
    initAuth();

    const unsubscribe = onAuthStateChanged(auth, (currentUser) => {
      setUser(currentUser);
      if (currentUser) {
        if (localStorage.getItem('quiro_registered')) setIsBiometricNeeded(true);
      } else {
        setIsBiometricNeeded(false);
      }
      setLoading(false);
      setAuthInProcess(false);
    });
    return () => unsubscribe();
  }, []);

  const handleInitialLogin = async () => {
    setAuthInProcess(true);
    try {
      if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
        try {
          await signInWithCustomToken(auth, __initial_auth_token);
        } catch (e) {
          await signInAnonymously(auth);
        }
      }
      else {
        await signInAnonymously(auth);
      }
      localStorage.setItem('quiro_registered', 'true');
    } catch (err) { setAuthInProcess(false); }
  };

  useEffect(() => {
    if (!user || isBiometricNeeded) return;
    
    getDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'settings', 'profile'))
      .then(snap => { 
        if (snap.exists()) setDoctorInfo(snap.data()); 
      })
      .catch(err => console.warn("No se pudo cargar el perfil, es posible que el dispositivo esté offline.", err));
    
    const unsubPat = onSnapshot(collection(db, 'artifacts', appId, 'users', user.uid, 'patients'), (snap) => {
      setPatients(snap.docs.map(d => ({ id: d.id, ...d.data() })));
    }, (err) => console.warn("Modo offline: Error al leer pacientes", err));
    
    const unsubApp = onSnapshot(collection(db, 'artifacts', appId, 'users', user.uid, 'appointments'), (snap) => {
      setAppointments(snap.docs.map(d => ({ id: d.id, ...d.data() })));
    }, (err) => console.warn("Modo offline: Error al leer citas", err));

    return () => { unsubPat(); unsubApp(); };
  }, [user, isBiometricNeeded]);

  const verifyAndExecute = (action) => {
    if (!doctorInfo.modPassword) action();
    else { setPendingAction(() => action); setModals(prev => ({ ...prev, password: true })); }
  };

  const handleSaveSettings = async (newData) => {
    if (!user) return;
    try {
      setDoctorInfo(newData);
      setModals(prev => ({ ...prev, settings: false }));
      await setDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'settings', 'profile'), newData);
    } catch(e) { console.error("Error al guardar ajustes", e); }
  };

  const handleAddPatient = async (data) => {
    if (!user) return;
    try {
      setModals(prev => ({ ...prev, patient: false }));
      await addDoc(collection(db, 'artifacts', appId, 'users', user.uid, 'patients'), { ...data, createdAt: new Date().toISOString() });
    } catch(e) { console.error("Error al agregar paciente", e); }
  };

  const handleAddAppointment = async (data) => {
    if (!user) return;
    try {
      setModals(prev => ({ ...prev, appointment: false }));
      setPrefilledPatientId("");
      await addDoc(collection(db, 'artifacts', appId, 'users', user.uid, 'appointments'), data);
    } catch(e) { console.error("Error al agregar cita", e); }
  };

  const handleAddHistory = async (patientId, historyData) => {
    if (!user) return;
    try {
      setModals(prev => ({ ...prev, history: false }));
      const patRef = doc(db, 'artifacts', appId, 'users', user.uid, 'patients', patientId);
      const patData = patients.find(p => String(p.id) === String(patientId));
      if(patData) {
        await updateDoc(patRef, { histories: [{ ...historyData, id: Date.now() }, ...(patData.histories || [])] });
      }
    } catch(e) { console.error("Error al guardar historial", e); }
  };

  const handleDeletePatient = async (patientId) => {
    if (!user) return;
    try {
      setSelectedPatientId(null);
      await deleteDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'patients', patientId));
      const appointmentsToDelete = appointments.filter(a => String(a.patientId) === String(patientId));
      for (const appt of appointmentsToDelete) {
         await deleteDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'appointments', appt.id));
      }
    } catch(e) { console.error("Error al eliminar paciente", e); }
  };

  if (showIntro) return <IntroScreen onFinish={() => setShowIntro(false)} />;
  if (loading) return <LoadingSpinner />;
  if (!user) return <AuthScreen onLogin={handleInitialLogin} inProcess={authInProcess} />;
  if (isBiometricNeeded) return <BiometricLockScreen onUnlocked={() => setIsBiometricNeeded(false)} />;

  const renderContent = () => {
    if (selectedPatientId) {
      return (
        <PatientProfile 
          patient={patients.find(p => String(p.id) === String(selectedPatientId))} 
          onBack={() => setSelectedPatientId(null)}
          onAddHistory={() => verifyAndExecute(() => setModals(prev => ({ ...prev, history: true })))}
          onDelete={() => verifyAndExecute(() => handleDeletePatient(selectedPatientId))}
          onScheduleAppointment={(pid) => {
            setPrefilledPatientId(pid);
            setModals(prev => ({ ...prev, appointment: true }));
          }}
        />
      );
    }
    switch (activeTab) {
      case 'home': 
        return <HomeTab appointments={appointments} patients={patients} doctorName={doctorInfo.name} clinicName={doctorInfo.clinic} onGoToAppointments={() => setActiveTab('appointments')} onSendReminder={(app) => openWhatsApp(patients.find(p => String(p.id) === String(app.patientId))?.phone, `Recuerde su cita hoy con Dr. ${doctorInfo.name}`)} />;
      case 'appointments': 
        return <AppointmentsTab appointments={appointments} patients={patients} doctorName={doctorInfo.name} onAddAppointment={() => { setPrefilledPatientId(""); setModals(prev => ({ ...prev, appointment: true })); }} onSendReminder={(app) => openWhatsApp(patients.find(p => String(p.id) === String(app.patientId))?.phone, `Recuerde su cita hoy con Dr. ${doctorInfo.name}`)} />;
      case 'patients': 
        return <PatientsTab patients={patients} onSelectPatient={setSelectedPatientId} onAddPatient={() => setModals(prev => ({ ...prev, patient: true }))} />;
      case 'marketing': 
        return <MarketingTab doctorInfo={doctorInfo} />;
      default: 
        return <HomeTab appointments={appointments} patients={patients} doctorName={doctorInfo.name} clinicName={doctorInfo.clinic} onGoToAppointments={() => setActiveTab('appointments')} onSendReminder={(app) => openWhatsApp(patients.find(p => String(p.id) === String(app.patientId))?.phone, `Recuerde su cita hoy con Dr. ${doctorInfo.name}`)} />;
    }
  };

  return (
    <div className="flex flex-col h-screen bg-[#020617] font-sans text-white animate-fade-in relative overflow-hidden">
      <SpineWatermark />
      
      <header className="bg-indigo-900/80 backdrop-blur-xl text-white p-4 shadow-2xl safe-top flex justify-between items-center z-50 border-b border-indigo-400/20">
        <div className="flex items-center gap-3">
          <SpineLogo className="w-8 h-8 sm:w-9 sm:h-9 text-cyan-400 drop-shadow-[0_0_8px_rgba(34,211,238,0.8)]" />
          <h1 className="text-xl sm:text-2xl font-black tracking-tighter uppercase italic hidden min-[400px]:block text-white">Quiro<span className="text-cyan-400 font-bold text-white">App</span></h1>
        </div>
        <div className="flex items-center gap-2 px-3 py-1.5 bg-white/5 rounded-full border border-white/10">
          <Cloud className={`w-3 h-3 sm:w-4 sm:h-4 ${isOnline ? 'text-cyan-400' : 'text-rose-500 animate-pulse'}`} />
          <span className="text-[8px] sm:text-[9px] font-black uppercase tracking-widest hidden sm:block">{isOnline ? 'CONECTADO' : 'SIN RED'}</span>
        </div>
        <div className="flex items-center gap-2 sm:gap-3">
          <button onClick={() => setModals(prev => ({ ...prev, settings: true }))} className="p-2 sm:p-2.5 bg-indigo-800/50 rounded-xl active:scale-90 transition border border-white/5"><Settings className="w-4 h-4 sm:w-5 sm:h-5 text-white" /></button>
        </div>
      </header>

      <main className="flex-1 overflow-y-auto pb-28 sm:pb-32 z-10 max-w-4xl mx-auto w-full px-2 sm:px-6">
        {renderContent()}
      </main>

      {!selectedPatientId && (
        <nav className="fixed bottom-0 w-full bg-indigo-950/90 backdrop-blur-2xl border-t border-indigo-400/30 flex justify-around py-3 sm:py-5 pb-safe z-40">
          <div className="flex justify-around w-full max-w-4xl mx-auto">
            <NavItem icon={<Home />} label="Inicio" isActive={activeTab === 'home'} onClick={() => setActiveTab('home')} />
            <NavItem icon={<CalendarIcon />} label="Agenda" isActive={activeTab === 'appointments'} onClick={() => setActiveTab('appointments')} />
            <NavItem icon={<Users />} label="Pacientes" isActive={activeTab === 'patients'} onClick={() => setActiveTab('patients')} />
            <NavItem icon={<Megaphone />} label="Marketing" isActive={activeTab === 'marketing'} onClick={() => setActiveTab('marketing')} />
          </div>
        </nav>
      )}

      {/* Modales Compartidos */}
      {modals.password && <PasswordPromptModal correctPassword={doctorInfo.modPassword} onSuccess={() => { if(pendingAction) pendingAction(); setModals(prev => ({ ...prev, password: false })); setPendingAction(null); }} onClose={() => { setModals(prev => ({ ...prev, password: false })); setPendingAction(null); }} />}
      {modals.appointment && <NewAppointmentModal patients={patients} initialPatientId={prefilledPatientId} onClose={() => { setModals(prev => ({ ...prev, appointment: false })); setPrefilledPatientId(""); }} onSave={handleAddAppointment} />}
      {modals.patient && <NewPatientModal onClose={() => setModals(prev => ({ ...prev, patient: false }))} onSave={handleAddPatient} />}
      {modals.history && selectedPatientId && <NewHistoryModal onClose={() => setModals(prev => ({ ...prev, history: false }))} onSave={(data) => handleAddHistory(selectedPatientId, data)} />}
      {modals.settings && <SettingsModal currentInfo={doctorInfo} onClose={() => setModals(prev => ({ ...prev, settings: false }))} onSave={handleSaveSettings} onLogout={handleLogout} />}
    </div>
  );
}
